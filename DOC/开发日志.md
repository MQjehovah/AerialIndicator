void flop_init1(void)
{
	P1OUT |= BIT1 + BIT2 + BIT3 + BIT6 + BIT7;
}
/*******************************************************************************
  * @brief  翻牌保持       
  * @param  None              
  * @retval None              
  * @Note   None              
*******************************************************************************/
void flop_hold(void)
{
	P1OUT |= BIT3 + BIT6;
	P1OUT &= ~BIT7;
	
	P1OUT &= ~BIT1;
	P1OUT &= ~BIT2;

}

/*******************************************************************************
  * @brief  反翻时3个引脚动作     
  * @param  None              
  * @retval None              
  * @Note   None              
*******************************************************************************/
void flop_do()
{
	P1OUT |= BIT7;
	P1OUT &= ~BIT3;
	P1OUT &= ~BIT6;	
}

/*******************************************************************************
  * @brief  翻红          
  * @param  None              
  * @retval None              
  * @Note   None              
*******************************************************************************/
void fanhong()
{

	delay_ms(100);

	delay_ms(100);

}
